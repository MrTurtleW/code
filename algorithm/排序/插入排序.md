遍历数组，将元素插入到前面已排好序的子数组中

# python

```python
import random

class InsertionSort:
    def __init__(self, numOfArray=10):
        self._lists = [random.randint(1, 100*numOfArray) for i in range(numOfArray)]

    def insert_element(self, elementIndex):
        for i in range(elementIndex):
            if self._lists[i] >= self._lists[elementIndex]:
                self._lists.insert(i, self._lists[elementIndex])
                del self._lists[elementIndex+1]
                break

    def insertion_sort(self):
        for i in range(1, len(self._lists)):
            self.insert_element(i)

    def print(self):
        for item in self._lists:
            print(item, end=', ')

        print()

if __name__ == '__main__':
    insertionSort = InsertionSort(100)
    insertionSort.print()
    insertionSort.insertion_sort()
    insertionSort.print()
```

```c
#include<stdio.h>
#include<stdlib.h>

#define NUM_OF_ARRAY 10

int move_element(int* array, int element_index){
	int i;
	for(i=element_index - 1; i>=0; i--){
		if(array[i] >= array[i+1]){
			int temp = array[i+1];
			array[i+1] = array[i];
			array[i] = temp;
		}
	}
}

int insertion_sort(int* array, int length){
	int i; 
	for(i=1;i<length; i++){
		move_element(array, i);
	}
}

void print_array(int* array, int length){
	int i;
	for(i=0;i<length; i++){
		printf("%d, ", array[i]); 
	}
	printf("\n");
}

int main(){
	int array[NUM_OF_ARRAY] = {0};
	int i;
	for(i=0; i<NUM_OF_ARRAY; i++){
		array[i] = rand() % (NUM_OF_ARRAY * 10) + 1; 
	}
	print_array(array, NUM_OF_ARRAY);
	insertion_sort(array, NUM_OF_ARRAY);
	print_array(array, NUM_OF_ARRAY);
}
```
